import 'package:flutter/foundation.dart';
import 'auth_service.dart';
import '../../models/user.dart';
import '../../models/organization.dart';
import '../api/api_client.dart';
import '../storage/storage_service.dart';

class AuthServiceImpl implements AuthService {
  final ApiClient _api;
  final StorageService _storage;

  User? _currentUser;
  String? _jwt;
  Organization? _selectedOrg;

  AuthServiceImpl(this._api, this._storage);

  @override
  Future<void> initialize() async {
    _jwt = await _storage.readToken();
    final orgId = await _storage.readSelectedOrganizationId();
    if (orgId != null) {
      _selectedOrg = Organization(id: orgId, name: '');
    }
  }

  @override
  Future<bool> signInWithGoogle() async {
    // Placeholder - real implementation should call GoogleSignIn
    // and then exchange token with backend
    debugPrint('AuthServiceImpl.signInWithGoogle called (stub)');
    // Simulate sign-in
    _jwt = 'fake-jwt-token';
    _currentUser = User(id: '1', email: 'demo@example.com', name: 'Demo User');
    await _storage.saveToken(_jwt!);
    return true;
  }

  @override
  Future<void> logout() async {
    _jwt = null;
    _currentUser = null;
    _selectedOrg = null;
    await _storage.clearToken();
    await _storage.clearSelectedOrganization();
  }

  @override
  bool get isLoggedIn => _currentUser != null && _jwt != null;

  @override
  bool get hasSelectedOrganization => _selectedOrg != null;

  @override
  String? get jwtToken => _jwt;

  @override
  User? get currentUser => _currentUser;

  @override
  Organization? get selectedOrganization => _selectedOrg;

  @override
  Future<void> selectOrganization(Organization org) async {
    _selectedOrg = org;
    await _storage.saveSelectedOrganizationId(org.id);
  }
}
